* common db
(c1) file -> ids
    "O" + "|" + fileName
    -> cuId + "," + dbId

* usr db
(u1) usr -> dbId(def)
    "T" + "|" + USR
    -> [dbId]+

(u2) usr ->  dbId(ref)
    "B" + "|" + USR
    -> [dbId]+

(u3) usr ->  dbId(overrider)
    "V" + "|" + USR
    -> [dbId]+

* main db
(1) position -> localUsrId
    dbId + "A" + "|" + lineNo + "|" + colNo
    -> localUsrId

(2) localUsrId -> decl
    dbId + "D" + "|" + localUsrId
    -> nameId + "|" + lineNo + "|" + colNo

(3) dbId -> cuFileList
    dbId + "S"
    -> [dbId]+

(4) usr -> def
    dbId + "E" + "|" + usr
    -> nameId + "," + lineNo + "," +colNo

(5) localUsrId -> usr
    dbId + "F" + "|" + localUsrId
    -> usr

(6) usr -> localUsrId
    dbId + "R" + "|" + usr
    -> localUsrId

(7) usr -> ref
    dbId + "L" + "|" + usr
    -> [namId + "," + lineNo + "," +colNo]+

(8) usr -> override
    dbId + "N" + "|" + usr
    -> [namId + "," + lineNo + "," +colNo]+

(9) usr -> overriden
    dbId + "M" + "|" + usr
    -> [namId + "," + lineNo + "," +colNo]+

(10) usr -> overriden
    dbId + "K" + "|" + usr
    -> filename + "," + workDir + "," + exclude + "," + opt
